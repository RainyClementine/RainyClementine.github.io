#include<bits/stdc++.h>
using namespace std;

const int N=100010;
const int M=300010;
typedef long long ll;
int n,m,q,tot,cnt,head[N],id[N],train[N],fa[N],f[N][26],d[N];
ll c[N],w[N][26];
struct Node{
	int x,y;
	ll z;
	Node(){}
	Node(int _x,int _y,ll _z):x(_x),y(_y),z(_z){}
	inline bool operator <(const Node& that)const{
		return z>that.z;
	} 
}a[M];

struct Edge{
	int suiv,ver;
	ll edge;
}e[M<<1];

inline int find(int x){return x==fa[x]?x:fa[x]=find(fa[x]);}
inline void lnk(int x,int y,ll z)
{
	e[++tot].ver=y;
	e[tot].edge=z;
	e[tot].suiv=head[x];
	head[x]=tot;
}

inline void dfs(int x)
{
	d[x]=d[f[x][0]]+1;
	for(int i=1;1<<i<=d[x];i++)
		f[x][i]=f[f[x][i-1]][i-1],
		w[x][i]=min(w[x][i-1],w[f[x][i-1]][i-1]);
	for(int i=head[x];i;i=e[i].suiv)
	{
		int y=e[i].ver;
		ll z=e[i].edge;
		if(y==f[x][0])continue;
		f[y][0]=x;
		w[y][0]=z;
		dfs(y);
	}
}

inline ll lca(int x,int y)
{
	ll res=1ll<<60;
	if(d[x]<d[y])swap(x,y);
	for(int i=25;~i;i--)
		if(d[f[x][i]]>=d[y])
			res=min(res,w[x][i]),x=f[x][i];
	if(x==y)return res;
	for(int i=25;~i;i--)
		if(f[x][i]!=f[y][i])
			res=min(res,min(w[x][i],w[y][i])),
			x=f[x][i],y=f[y][i];
	return min(res,min(w[x][0],w[y][0]));
}

int main()
{
	scanf("%d%d%d",&n,&m,&q);
	for(int i=1;i<=n;i++)
		scanf("%d",&id[i]),fa[i]=i;
	for(int i=1;i<=n;i++)
		scanf("%lld",&c[i]);
	for(int i=1;i<=m;i++)
		scanf("%d%d%lld",&a[i].x,&a[i].y,&a[i].z);
	for(int i=1;i<=q;i++)
		scanf("%d",&train[i]);
	cnt=m;
	for(int i=2;i<=q;i++)
		a[++cnt]=Node(train[1],train[i],1ll<<60);
	sort(a+1,a+1+cnt);
	for(int i=1;i<=cnt;i++)
	{
		int fx=find(a[i].x),fy=find(a[i].y);
		if(fx==fy)continue;
		fa[fx]=fy;
		lnk(a[i].x,a[i].y,a[i].z);
		lnk(a[i].y,a[i].x,a[i].z);
	}
	dfs(1);
	ll pos=0;
	if(c[id[1]]>0)pos=c[id[1]];
	else puts("0");
	for(int i=1;i<n;i++)
	{
		ll maxi=lca(id[i],id[i+1]);
		pos=min(pos,maxi);
		if(c[id[i+1]]>0)
			pos+=c[id[i+1]];
		else printf("%lld\n",min(pos,-c[id[i+1]])),
			pos=max(0ll,pos+c[id[i+1]]);
	}
	return 0;
}
