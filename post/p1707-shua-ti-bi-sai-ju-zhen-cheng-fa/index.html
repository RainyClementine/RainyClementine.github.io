inline int fmul(int x,int y)
{
	int res=0;
	for(;y;y>>=1,x=(x<<1)%mod)
		if(y&1)(res+=x)%=mod;
	return res;
}

struct Matrix{
	int r,c,a[N][N];
	
	Matrix(){r=c=0,memset(a,0,sizeof(a));}
	Matrix(int _r,int _c):r(_r),c(_c)
		{memset(a,0,sizeof(a));}
	
	inline Matrix operator *(Matrix x){
		Matrix t=*this;
		Matrix res(t.r,x.c);
		for(int i=1;i<=res.r;i++)
			for(int k=1;k<=t.c;k++)
				for(int j=1;j<=res.c;j++)
					(res.a[i][j]+=fmul(t.a[i][k],x.a[k][j]))%=mod;
		return res;
	}
	
	inline void opt(){
		for(int i=1;i<=r;i++)

inline Matrix fpow(Matrix t,int x)
{
	Matrix res(t.r,t.r);
	for(int i=1;i<=res.r;i++)
		res.a[i][i]=1;
	for(;x;x>>=1,t=t*t)
		if(x&1)res=res*t;
	return res;
}
