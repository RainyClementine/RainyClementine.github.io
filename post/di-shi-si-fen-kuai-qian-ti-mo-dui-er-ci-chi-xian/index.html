第十四分块(前体) 莫队二次离线

2019-11-28

给定一个数列 多次询问(l,r)
每次询问al..ar 中ax xor ay有k位是1的二元组(x,y)个数

二次离线莫队
对于贡献的计算 因为ai值域很小
所以可以把所有有k位1的数字预处理出来 存在valid[]中
update(x) 表示加入x到集合中
query(x) 返回集合中有多少个元素和x异或的结果中有k位是1
有两种配套的update和query的写法

//1:
inline void update(int x){c[x]++;}
inline int query(int x)
{
	int res=0;
	for(int i=1;i<=cnt;i++)
		res+=c[x^valid[i]];
	return res;
}
//2:
inline void update(int x)
{
	for(int i=1;i<=cnt;i++)
		c[x^valid[i]]++;
}
inline int query(int x){return c[x];}

在插入和询问同级时 复杂度是一样的
但是因为二次离线时 需要进行多次询问
询问次数远远大于查询 所以要采取第二种写法

#pragma GCC optimize("Ofast")
#pragma GCC target("sse3","sse2","sse")
#pragma GCC target("avx","sse4","sse4.1","sse4.2","ssse3")
#pragma GCC target("f16c")
#pragma GCC target("fma","avx2")
#pragma GCC target("xop","fma4")
#pragma GCC optimize("inline","fast-math","unroll-loops","no-stack-protector")
#pragma GCC diagnostic error "-fwhole-program"
#pragma GCC diagnostic error "-fcse-skip-blocks"
#pragma GCC diagnostic error "-funsafe-loop-optimizations"
#include<bits/stdc++.h>
using namespace std;
#define getchar() (p1==p2&&(p2=(p1=buf)+fread(buf,1,1<<21,stdin),p1==p2)?EOF:*p1++)
char buf[1<<23],*p1=buf,*p2=buf,obuf[1<<23],*O=obuf;

const int N=100010;
const int M=35000;
typedef long long ll;
int n,m,k,cnt,block,bel[N];
ll a[N],ans[N],d1[N],d2[N],c[N],valid[N];

struct Query{
	int l,r,id;ll ans;
	inline bool operator <(const Query& that)const{
		return bel[l]==bel[that.l] ? bel[l]&1 ?
			r<that.r : r>that.r : l<that.l ;
	}
}q[N];
vector<Query>ml[N],mr[N];

inline int g()
{
	register ll res=0,fix=1;
	char ch;
	while(!isdigit(ch=getchar()))
		fix=(ch=='-')?-1:fix;
	do res=res*10+(ch^'0');
	while(isdigit(ch=getchar()));
	return fix*res;
}

inline void out(ll x)
{
	if(x<0)putchar('-'),x=-x;
	if(x>9)out(x/10);
	putchar(x%10+'0');
}

inline void update(int x)
{
	for(int i=1;i<=cnt;i++)
		c[x^valid[i]]++;
}

int main()
{
	n=g(),m=g(),k=g();
	for(register int i=0;i<16390;i++)
		if(__builtin_popcount(i)==k)valid[++cnt]=i;
	for(register int i=1;i<=n;i++)a[i]=g();
	for(register int i=1;i<=m;i++)
		q[i].l=g(),q[i].r=g(),q[i].id=i;
	block=(int)sqrt(n);
	for(register int i=1;i<=n;i++)
		bel[i]=(i-1)/block+1;
	for(register int i=1;i<=n;i++)d1[i]=c[a[i]],update(a[i]);
	memset(c,0,sizeof(c));
	for(register int i=n;i;i--)d2[i]=c[a[i]],update(a[i]);
	memset(c,0,sizeof(c));
	sort(q+1,q+1+m);
	int l=1,r=0;
	for(register int i=1,L,R;i<=m;i++)
	{
		L=q[i].l,R=q[i].r;
		if(r<R)ml[l-1].push_back((Query){r+1,R,i,-1});
		while(r<R)q[i].ans+=d1[++r];
		if(r>R)ml[l-1].push_back((Query){R+1,r,i,1});
		while(r>R)q[i].ans-=d1[r--];
		if(l<L)mr[r+1].push_back((Query){l,L-1,i,1});
		while(l<L)q[i].ans-=d2[l++];
		if(l>L)mr[r+1].push_back((Query){L,l-1,i,-1});
		while(l>L)q[i].ans+=d2[--l];
	}
	for(register int i=1;i<=n;i++)
	{
		update(a[i]);
		for(auto tem:ml[i])
			for(int j=tem.l;j<=tem.r;j++)
				q[tem.id].ans+=tem.ans*c[a[j]]; 
	}
	memset(c,0,sizeof(c));
	for(register int i=n;i;i--)
	{
		update(a[i]);
		for(auto tem:mr[i])
			for(int j=tem.l;j<=tem.r;j++)
				q[tem.id].ans+=tem.ans*c[a[j]];
	}
	for(register int i=1;i<=m;i++)
		q[i].ans+=q[i-1].ans;
	for(register int i=1;i<=m;i++)
		ans[q[i].id]=q[i].ans;
	for(register int i=1;i<=m;i++)out(ans[i]),putchar('\n');
	return 0;
}
